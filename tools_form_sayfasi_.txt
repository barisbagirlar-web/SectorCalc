function refStrip(f){
  if(f.type==="enum"||f.type==="bool") return `<div class="ref"><span class="info" tabindex="0">guidance<span class="tip">${f.note}</span></span></div>`;
  const segs=[];
  if(f.type==="ratio"){
    segs.push(`<span class="seg">range <b>${f.min*100}–${(f.max*100)}%</b></span>`);
    segs.push(`<span class="seg">ref <b>${(f.def*100).toFixed(f.step<0.01?1:0)}%</b></span>`);
  }else{
    segs.push(`<span class="seg">range <b>${fmtN(f.min,f.step<1?1:0)}–${f.max>=1e7?"∞":fmtN(f.max,0)}</b></span>`);
    segs.push(`<span class="seg">step <b>${f.step}</b></span>`);
    if(f.ref) segs.push(`<span class="seg">ref <b>${f.ref}</b></span>`);
    else if(f.def!==null&&f.def!==0) segs.push(`<span class="seg">ref <b>${fmtN(f.def,f.step<1?1:0)}</b></span>`);
  }
  return `<div class="ref">${segs.join('<span class="dot"></span>')}<span class="dot"></span><span class="info" tabindex="0">i<span class="tip">${f.note}</span></span><span class="pct" id="pct_${f.id}"></span></div>`;
}

/* ---------- validation (V1..V17, fail-closed) ---------- */
function validate(){
  const e={}; const s=state; const num=id=>Number(s[id]);
  const req=["shipment_units","package_count","package_length_cm","package_width_cm","package_height_cm","gross_weight_total_kg","volumetric_divisor_cm3_per_kg","product_value_total","freight_rate_per_chargeable_kg"];
  req.forEach(id=>{ if(s[id]===null||s[id]===undefined||s[id]==="") e[id]="Required."; });
  if(!e.shipment_units && (num("shipment_units")<1||!Number.isInteger(num("shipment_units")))) e.shipment_units="Must be a positive integer.";
  if(!e.package_count && (num("package_count")<1||!Number.isInteger(num("package_count")))) e.package_count="Must be a positive integer.";
  ["package_length_cm","package_width_cm","package_height_cm"].forEach(id=>{ if(!e[id]&&num(id)<=0)e[id]="Must be positive."; });
  if(!e.gross_weight_total_kg && num("gross_weight_total_kg")<=0) e.gross_weight_total_kg="Must be positive.";
  if(!e.volumetric_divisor_cm3_per_kg && num("volumetric_divisor_cm3_per_kg")<=0) e.volumetric_divisor_cm3_per_kg="Must be positive.";
  if(num("chargeable_rounding_increment_kg")<=0) e.chargeable_rounding_increment_kg="Increment must be positive.";
  if(num("minimum_chargeable_weight_kg")<0) e.minimum_chargeable_weight_kg="Cannot be negative.";
  if(!e.product_value_total && num("product_value_total")<0) e.product_value_total="Cannot be negative.";
  ["freight_rate_per_chargeable_kg","flat_freight_charge","minimum_freight_charge","fuel_surcharge_rate","security_surcharge_per_kg","other_freight_surcharge_total","origin_local_charges","destination_local_charges","origin_inland_transport_cost","destination_inland_transport_cost","insurance_rate","insurance_uplift_rate","duty_rate","import_tax_rate","customs_brokerage_fee","storage_demurrage_detention_cost","bank_payment_fee","target_selling_price_per_unit"].forEach(id=>{ if(s[id]!==null&&num(id)<0)e[id]="Cannot be negative."; });
  if(num("minimum_margin_rate")<0||num("minimum_margin_rate")>=1) e.minimum_margin_rate="Must be 0 to <1.";
  return e;
}

/* ---------- compute F1..F30 ---------- */
function compute(){
  const s=state, n=id=>Number(s[id]);
  const F={};
  F.package_volume_cm3 = n("package_length_cm")*n("package_width_cm")*n("package_height_cm");
  F.total_volume_cm3 = F.package_volume_cm3*n("package_count");
  F.total_volume_m3 = F.total_volume_cm3/1e6;
  F.volumetric_weight_kg = F.total_volume_cm3/n("volumetric_divisor_cm3_per_kg");
  F.shipment_density_kg_per_m3 = n("gross_weight_total_kg")/Math.max(1e-9,F.total_volume_m3);
  F.raw_chargeable_weight_kg = Math.max(n("gross_weight_total_kg"),F.volumetric_weight_kg,n("minimum_chargeable_weight_kg"));
  const R=n("chargeable_rounding_increment_kg");
  F.chargeable_weight_kg = Math.ceil(F.raw_chargeable_weight_kg/R)*R;
  F.chargeable_weight_driver = F.raw_chargeable_weight_kg===n("minimum_chargeable_weight_kg") ? "MINIMUM_WEIGHT"
      : (F.volumetric_weight_kg>n("gross_weight_total_kg")?"VOLUMETRIC_WEIGHT":"GROSS_WEIGHT");
  F.weight_based_freight_charge = F.chargeable_weight_kg*n("freight_rate_per_chargeable_kg");
  F.base_freight_charge = Math.max(n("minimum_freight_charge"),F.weight_based_freight_charge+n("flat_freight_charge"));
  F.fuel_surcharge_cost = F.base_freight_charge*n("fuel_surcharge_rate");
  F.security_surcharge_cost = F.chargeable_weight_kg*n("security_surcharge_per_kg");
  F.total_freight_cost = F.base_freight_charge+F.fuel_surcharge_cost+F.security_surcharge_cost+n("other_freight_surcharge_total");
  F.insurance_value_basis = (n("product_value_total")+F.total_freight_cost)*(1+n("insurance_uplift_rate"));
  F.insurance_cost = F.insurance_value_basis*n("insurance_rate");
  F.customs_value_basis = Math.max(0,n("product_value_total")+F.total_freight_cost+F.insurance_cost+n("customs_value_adjustment"));
  F.customs_duty_cost = F.customs_value_basis*n("duty_rate");
  F.import_tax_base = F.customs_value_basis+F.customs_duty_cost+n("customs_brokerage_fee")+n("destination_local_charges");
  F.import_tax_cash_cost = F.import_tax_base*n("import_tax_rate");
  F.economic_import_tax_cost = s.tax_recoverable?0:F.import_tax_cash_cost;
  F.total_local_admin_charges = n("origin_local_charges")+n("destination_local_charges")+n("origin_inland_transport_cost")+n("destination_inland_transport_cost")+n("customs_brokerage_fee")+n("storage_demurrage_detention_cost")+n("bank_payment_fee");
  F.total_landed_cost = n("product_value_total")+F.total_freight_cost+F.insurance_cost+F.customs_duty_cost+F.economic_import_tax_cost+F.total_local_admin_charges;
  F.landed_cost_per_unit = F.total_landed_cost/n("shipment_units");
  F.freight_cost_per_unit = F.total_freight_cost/n("shipment_units");
  F.landed_to_goods_value_ratio = n("product_value_total")>0?F.total_landed_cost/n("product_value_total"):null;
  F.target_selling_revenue = n("target_selling_price_per_unit")*n("shipment_units");
  F.gross_margin_after_landed_cost = n("target_selling_price_per_unit")>0?(F.target_selling_revenue-F.total_landed_cost)/F.target_selling_revenue:null;
  F.required_selling_price_per_unit = n("minimum_margin_rate")<1?F.landed_cost_per_unit/(1-n("minimum_margin_rate")):null;
  F.margin_gap = F.gross_margin_after_landed_cost===null?null:F.gross_margin_after_landed_cost-n("minimum_margin_rate");
  F.decision = F.gross_margin_after_landed_cost===null?"COST_ONLY_NO_SELLING_PRICE"
      : (F.gross_margin_after_landed_cost>=n("minimum_margin_rate")?"LANDED_COST_ACCEPTABLE":"PRICE_OR_FREIGHT_REVIEW_REQUIRED");
  return F;
}

/* ---------- smart warnings W1..W9 ---------- */
function warnings(F){
  const s=state,n=id=>Number(s[id]); const out=[];
  if(F.chargeable_weight_driver==="VOLUMETRIC_WEIGHT") out.push(["WARNING","Volumetric weight controls freight cost. Review packaging density and carton/pallet dimensions."]);
  if(F.volumetric_weight_kg>n("gross_weight_total_kg")*2) out.push(["WARNING","Volumetric weight exceeds 2× gross weight. Low-density shipment — packaging optimization may cut freight cost."]);
  if(s.freight_mode==="SEA_FCL"&&n("volumetric_divisor_cm3_per_kg")===6000) out.push(["WARNING","Sea FCL is usually costed per container, not with an air-style volumetric divisor. Verify the charge basis."]);
  if(s.incoterm==="DDP"&&(F.customs_duty_cost+F.import_tax_cash_cost)>0) out.push(["WARNING","DDP normally assigns duty/tax to the seller. Confirm whether the buyer should carry customs charges."]);
  if(s.tax_recoverable&&F.import_tax_cash_cost>0) out.push(["INFO","Import tax is recoverable and excluded from economic landed cost, but the cash outflow is still reported."]);
  if(F.landed_to_goods_value_ratio!==null&&F.landed_to_goods_value_ratio>1.5) out.push(["WARNING","Landed cost exceeds 150% of goods value. Review freight mode, duty rate, local charges and consolidation."]);
  if(F.gross_margin_after_landed_cost!==null&&F.gross_margin_after_landed_cost<n("minimum_margin_rate")) out.push(["CRITICAL","Gross margin after landed cost is below threshold. Raise price, renegotiate freight, consolidate, or change sourcing."]);
  if(n("storage_demurrage_detention_cost")>F.total_freight_cost*0.25&&F.total_freight_cost>0) out.push(["WARNING","Storage/demurrage/detention is high versus freight. Investigate customs delay, free time and document readiness."]);
  if(n("duty_rate")>0.25) out.push(["WARNING","High duty rate. Verify HS code, origin preference, anti-dumping exposure and classification."]);
  return out;
}
