function cncCalc(){
  const i = cncInp;
  const errs = [];
  const D=+i.D, z=+i.z, Vc=+i.Vc, fz=+i.fz, re=+i.re;
  const ap=+i.ap, ae=+i.ae, kc1=+i.kc1, mc=+i.mc, eta=+i.eta;
  const Pmax=+i.Pmax, nmax=+i.RPMmax||12000;

  if(!D||D<0.1) errs.push({id:'V0',msg:'Tool diameter (D) is required (min 0.1 mm).'});
  if(!z||z<1)   errs.push({id:'V0',msg:'Flute count (z) must be at least 1.'});
  if(!Vc||Vc<5) errs.push({id:'V0',msg:'Cutting speed (V_c) minimum 5 m/min required.'});
  if(!fz||fz<0.001) errs.push({id:'V2',msg:'Feed per tooth (f_z) minimum 0.001 mm/tooth — V2 BLOCK.'});
  if(!re||re<0.01)  errs.push({id:'V4',msg:'Nose radius (r_ε) minimum 0.01 mm — V4 BLOCK.'});
  if(!ap||ap<=0) errs.push({id:'V0',msg:'Axial depth (a_p) must be positive.'});
  if(!ae||ae<=0) errs.push({id:'V0',msg:'Radial width (a_e) must be positive.'});
  if(!kc1||kc1<100) errs.push({id:'V0',msg:'k_c1 is required. Select material group or enter manually.'});
  if(!mc||mc<0.05) errs.push({id:'V0',msg:'Kienzle exponent m_c required (typical 0.14–0.35).'});
  if(!eta||eta<50) errs.push({id:'V0',msg:'Spindle efficiency (η) required (min 50%).'});
  if(!Pmax||Pmax<1) errs.push({id:'V0',msg:'Machine rated power (P_mach) required.'});
  if(D&&ae&&ae>D) errs.push({id:'V1',msg:'a_e > D: Radial width exceeds tool diameter — V1 BLOCK. Geometrically impossible.'});
  if(D&&ap&&ap>D*0.8) errs.push({id:'V3',msg:'a_p > 0.8D: Axial depth exceeds 80% of diameter — V3 BLOCK. Catastrophic load risk.'});

  const ep = document.getElementById('cnc-err-panel');
  if(errs.length){
    ep.style.display='block';
    ep.innerHTML=errs.map(e=>`<div class="err-item"><span class="err-code">${e.id}</span><span class="err-msg">${e.msg}</span></div>`).join('');
    return;
  }
  ep.style.display='none';

  // ── CALCULATIONS ──
  const n_rpm_calc = (1000*Vc)/(Math.PI*D);
  const n_act = Math.min(n_rpm_calc, nmax);
  const Vc_act = (n_act*Math.PI*D)/1000;
  const Vf = fz*z*n_act;
  const h_m = fz*Math.sqrt(ae/D);
  const kc_act = kc1/Math.pow(h_m, mc);
  const Fc = kc_act*ap*fz;
  const Pc_net = (Fc*Vc_act)/60000;
  const P_motor = Pc_net/(eta/100);
  const MRR = (ap*ae*Vf)/1000;
  const Rz = (fz*fz/(8*re))*1000;
  const Ra = Rz/4.5;
  let T_life = null;
  if(cncActiveMat){const m=MAT_DB[cncActiveMat];T_life=Math.pow(m.C/Vc_act,1/m.n);}

  // GUM uncertainty (Type B, relative)
  const u_D=0.01/D, u_fz=0.03, u_kc1=0.05, u_mc=0.02;
  const u_Fc = Fc*Math.sqrt(u_kc1**2+(u_fz*mc)**2);
  const u_P = P_motor*Math.sqrt(u_D**2+u_fz**2+u_kc1**2);

  // UC analogous: power utilization
  const UC_power = P_motor/Pmax;

  // ── WARNINGS ──
  const warns = [];
  if(cncActiveMat){

Chip adhesion, surface deterioration, and tool fracture are expected. Increase V_c to ${m.vc[0]}–${m.vc[2]} m/min range.`});

    if(T_life!==null&&T_life<15) warns.push({sev:'WARNING',id:'W3',src:'ISO 3685 Taylor',msg:`Short Tool Life: T_life ≈ ${T_life.toFixed(1)} min. Industry minimum target is 30 min. Reduce V_c toward optimum ${m.vc[1]} m/min.`});
  }
  if(P_motor > Pmax) warns.push({sev:'CRITICAL',id:'W4',src:'Machine Limit',msg:`Power Overload: P_motor = ${P_motor.toFixed(2)} kW exceeds P_machine = ${Pmax} kW. Spindle stall and tool crash are expected. Reduce a_p or a_e.`});
  if(Ra > 3.2) warns.push({sev:'WARNING',id:'W5',src:'ISO 4287 / ISO 1302',msg:`Surface Quality: R_a = ${Ra.toFixed(2)} μm exceeds Ra ≤ 3.2 μm general limit. Reduce f_z or use wiper-geometry insert.`});

    {name:'Theoretical Peak Roughness',sym:'R_z',v:Rz,u:'μm',d:2},
    {name:'Arithmetic Mean Roughness',sym:'R_a',v:Ra,u:'μm',d:2,hl:true},
    {name:'Taylor Tool Life',sym:'T_life',v:T_life,u:'min',d:1},

function weldApplyPreset(k){

}

function weldCalc(){
  const i = weldInp;
  const errs = [];
  const C=+(i.c||0), Mn=+(i.mn||0), Cr=+(i.cr||0), Mo=+(i.mo||0);
  const V=+(i.v||0), Ni=+(i.ni||0), Cu=+(i.cu||0);
  const t=+(i.t||0), H=+(i.H||0);

  if(i.c===undefined||i.c===null||i.c==='') errs.push({id:'V0',msg:'Carbon content (C) is required.'});
  if(!t||t<1) errs.push({id:'V2',msg:'Section thickness (t) minimum 1 mm — V2 BLOCK.'});
  if(C>2.11) errs.push({id:'V1',msg:'C > 2.11% — Material exits steel classification (cast iron range). CE_IIW method is invalid — V1 BLOCK.'});

  const ep = document.getElementById('weld-err-panel');
  if(errs.length){
    ep.style.display='block';
    ep.innerHTML=errs.map(e=>`<div class="err-item"><span class="err-code">${e.id}</span><span class="err-msg">${e.msg}</span></div>`).join('');
    return;
  }
  ep.style.display='none';

  const CE = C + Mn/6 + (Cr+Mo+V)/5 + (Ni+Cu)/15;
  let Tp_CE = 0;
  if(CE>=0.45&&CE<0.60) Tp_CE=100+(CE-0.45)*1000;
  else if(CE>=0.60) Tp_CE=200+(CE-0.60)*500;

  let Tp_t = 0;
  if(t>=20&&t<40) Tp_t=50;
  else if(t>=40&&t<60) Tp_t=100;
  else if(t>=60) Tp_t=150;

  const Tp_req = Math.max(Tp_CE, Tp_t) + H;

  // GUM uncertainty
  const u_C=0.01, u_Mn=0.05, u_Cr=0.05, u_Mo=0.02, u_V=0.01, u_Ni=0.05, u_Cu=0.02;
  const u_CE = Math.sqrt(u_C**2 + (u_Mn/6)**2 + ((u_Cr+u_Mo+u_V)/5)**2 + ((u_Ni+u_Cu)/15)**2);

  const warns=[];
  if(CE>=0.45&&Tp_req<100) warns.push({sev:'CRITICAL',id:'W3',src:'AWS D1.1:2020 / EN 1011-2',msg:`Cold Cracking Risk — MANDATORY PREHEAT: CE_IIW = ${CE.toFixed(4)} ≥ 0.45. Hydrogen-induced cracking and martensite formation in HAZ are certain without preheat. Required minimum: ${Tp_req}°C. Welding WITHOUT preheat violates AWS D1.1 §4.2.`});
  if(CE>=0.60) warns.push({sev:'CRITICAL',id:'W5',src:'IIW Doc. IX-2136 / ASME BPVC IX',msg:`High CE Zone: CE_IIW = ${CE.toFixed(4)} ≥ 0.60. Special WPS with multi-pass procedure, interpass temperature control, and post-weld heat treatment (PWHT) are mandatory per ASME BPVC IX QW-407.`});
  if(t>60&&CE>0.40) warns.push({sev:'WARNING',id:'W3b',src:'EN 1011-2 Table B.2',msg:`Heavy Section: t = ${t} mm with CE > 0.40. t8/5 cooling time calculation is required. PWHT evaluation is recommended per ISO 15614-1.`});
  if(CE<0.45&&Tp_req===0) warns.push({sev:'INFO',id:'W0',src:'AWS D1.1 §4.2',msg:`Low CE Group: CE_IIW = ${CE.toFixed(4)} < 0.45. Preheat is not mandated. However, if ambient temperature < 5°C, preheat to 20°C minimum per AWS D1.1 Table 4.5 note.`});
  if(Cr>12&&Ni>8) warns.push({sev:'WARNING',id:'W4',src:'AWS D1.1 Annex K / Schaeffler',msg:`Austenitic Stainless: Cr > 12% AND Ni > 8% detected. CE_IIW is NOT valid for this material class. Use CrEq/NiEq Schaeffler-Delong diagram for delta ferrite and solidification mode assessment.`});

  const wHtml = warns.map(w=>`<div class="warn-card ${w.sev}">
    <div class="warn-hdr"><span>${w.sev==='CRITICAL'?'⛔':'⚠'}</span>
    <span class="warn-sev">${w.sev}</span><span class="warn-id">${w.id}</span>
    <span class="warn-src">${w.src}</span></div>
    <div class="warn-msg">${w.msg}</div></div>`).join('');

  const riskLevel = CE>=0.60?'HIGH — PWHT mandatory':CE>=0.45?'ELEVATED — Preheat mandatory':'LOW — Standard procedure applicable';
  const riskCls = CE>=0.60?'fail':CE>=0.45?'warn':'ok';

  const rows=[
    {name:'CE_IIW (IIW Carbon Equivalent)',sym:'CE_IIW',v:CE,u:'—',d:4,hl:true},
    {name:'CE Uncertainty u_c(CE)',sym:'u_c(CE)',v:u_CE,u:'—',d:4},
    {name:'Preheat from CE Method',sym:'T_p,CE',v:Tp_CE,u:'°C',d:0},
    {name:'Preheat from Thickness',sym:'T_p,t',v:Tp_t,u:'°C',d:0},
    {name:'Hydrogen Correction',sym:'ΔT_H',v:H,u:'°C',d:0},
    {name:'Required Minimum Preheat',sym:'T_p,req',v:Tp_req,u:'°C',d:0,hl:true},
    {name:'Cold Crack Risk Assessment',sym:'HAZ',v:null,txt:riskLevel,cls:riskCls},
  ];

  const rHtml = rows.map(r=>{
    const dv = r.txt ? `<span class="res-val ${r.cls||''}">${r.txt}</span>` : `<span class="res-val${r.hl?' hl':''}">${r.v===null?'—':r.v.toFixed(r.d)}</span>`;
    return `<div class="res-row">
      <span class="res-name">${r.name}<span class="res-sym">${r.sym}</span></span>
      <span></span>
      ${dv}
      <span class="res-unit">${r.txt?'':r.u}</span>
    </div>`;
  }).join('');

  const gumHtml=`<div class="gum-table">
    <div class="gum-hdr"><span>Parameter</span><span>Uncertainty</span><span>Type</span><span>Contribution</span></div>
    <div class="gum-row"><span class="gum-param">C</span><span class="gum-unc">±0.010%</span><span class="gum-type">B</span><span class="gum-contrib">${(u_C*100/u_CE*100).toFixed(0)}%</span></div>
    <div class="gum-row"><span class="gum-param">Mn/6</span><span class="gum-unc">±${(u_Mn/6).toFixed(4)}%</span><span class="gum-type">B</span><span class="gum-contrib">${((u_Mn/6)/u_CE*100).toFixed(0)}%</span></div>
    <div class="gum-row"><span class="gum-param">(Cr+Mo+V)/5</span><span class="gum-unc">±${((u_Cr+u_Mo+u_V)/5).toFixed(4)}%</span><span class="gum-type">B</span><span class="gum-contrib">${(((u_Cr+u_Mo+u_V)/5)/u_CE*100).toFixed(0)}%</span></div>
    <div class="gum-row"><span class="gum-param">(Ni+Cu)/15</span><span class="gum-unc">±${((u_Ni+u_Cu)/15).toFixed(4)}%</span><span class="gum-type">B</span><span class="gum-contrib">${(((u_Ni+u_Cu)/15)/u_CE*100).toFixed(0)}%</span></div>
    <div class="gum-total"><span>u_c(CE_IIW)</span><span class="gum-unc">±${u_CE.toFixed(4)}</span><span></span><span class="gum-contrib">Combined</span></div>
  </div>`;

  const verdict = warns.some(w=>w.sev==='CRITICAL')
    ? `<div class="verdict-banner fail"><span class="verdict-icon">⛔</span><div class="verdict-body"><div class="verdict-title">MANDATORY PREHEAT REQUIRED — Cold cracking risk is critical</div><div class="verdict-sub">CE_IIW = ${CE.toFixed(4)} · Required T_p,req = ${Tp_req}°C · Verify with welding engineer</div></div></div>`
    : CE>=0.40 ? `<div class="verdict-banner warn"><span class="verdict-icon">⚠</span><div class="verdict-body"><div class="verdict-title">ELEVATED RISK — Preheat recommended</div><div class="verdict-sub">CE_IIW = ${CE.toFixed(4)} · T_p,req = ${Tp_req}°C · AWS D1.1 §4.2 monitoring required</div></div></div>`
    : `<div class="verdict-banner pass"><span class="verdict-icon">✓</span><div class="verdict-body"><div class="verdict-title">LOW CE — Standard welding procedure applicable</div><div class="verdict-sub">CE_IIW = ${CE.toFixed(4)} · No mandatory preheat · u_c(CE) = ±${u_CE.toFixed(4)}</div></div></div>`;

  document.getElementById('weld-results-content').innerHTML =
    verdict +
    (warns.length ? `<div class="warn-panel">${wHtml}</div>` : '<div class="ok-banner">✓ CE_IIW below critical threshold. No mandatory preheat by CE criterion.</div>') +
    `<div class="sec-lbl">CE_IIW Analysis Results <span class="sec-note">EN 1011-2:2001 Method B · AWS D1.1:2020</span></div>` +
    `<div class="res-table"><div class="res-hdr"><span>Parameter</span><span></span><span>Value</span><span>Unit</span></div>${rHtml}</div>` +
    `<div class="sec-lbl">GUM Uncertainty Budget — u_c(CE_IIW) <span class="sec-note">ISO/IEC Guide 98-3 · Type B</span></div>${gumHtml}` +
    `<div class="refs-row"><span class="ref-ttl">References:</span><span class="ref-tag">AWS D1.1:2020</span><span class="ref-tag">EN 1011-2:2001</span><span class="ref-tag">IIW IX-2136</span><span class="ref-tag">ASME BPVC IX</span><span class="ref-tag">ISO 15614-1</span><span class="ref-tag">ISO/IEC 98-3</span></div>`;

  const crits=warns.filter(w=>w.sev==='CRITICAL').length;
  document.getElementById('weld-status').innerHTML = crits?`<span class="badge badge-crit">⛔ ${crits} CRITICAL</span>`:CE>=0.4?`<span class="badge badge-warn">⚠ ELEVATED RISK</span>`:`<span class="badge badge-ok">✓ LOW CE</span>`;
  document.getElementById('weld-uc-display').innerHTML=`CE_IIW: <strong>${CE.toFixed(4)}</strong> &nbsp;±${u_CE.toFixed(4)}`;
  if(warns.length) document.getElementById('weld-dot').style.display='inline-block';

  const ts=nowISO();
  const inputHash=hashStr(JSON.stringify(weldInp));
  document.getElementById('weld-audit-body').innerHTML =
    buildAuditRow('tool_id','PRO_091')+
    buildAuditRow('version','1.0.0')+
    buildAuditRow('timestamp',ts)+
    buildAuditRow('design_standard','EN 1011-2:2001 Method B / AWS D1.1:2020')+
    buildAuditRow('input_hash',inputHash)+
    buildAuditRow('C_%',C.toFixed(3))+
    buildAuditRow('Mn_%',Mn.toFixed(3))+
    buildAuditRow('Cr_%',Cr.toFixed(3))+
    buildAuditRow('Mo_%',Mo.toFixed(3))+
    buildAuditRow('V_%',V.toFixed(3))+
    buildAuditRow('Ni_%',Ni.toFixed(3))+
    buildAuditRow('Cu_%',Cu.toFixed(3))+
    buildAuditRow('thickness_mm',t)+
    buildAuditRow('H_correction_C',H)+
    buildAuditRow('CE_IIW',CE.toFixed(4))+
    buildAuditRow('u_c_CE','±'+u_CE.toFixed(4))+
    buildAuditRow('T_p_req_C',Tp_req)+
    buildAuditRow('cold_crack_risk',riskLevel,riskCls)+
    buildAuditRow('validation_result',crits>0?'FAIL — Mandatory preheat not met':'PASS — CE criterion satisfied',crits>0?'fail':'ok')+
    buildAuditRow('disclaimer','WPS qualification and NDT per ISO 17637 / EN 5817 required before production welding.');

  document.getElementById('weld-cbar-status').innerHTML=crits?`<span class="badge badge-crit">⛔ ${crits} CRITICAL</span>`:`<span class="badge badge-ok">✓ COMPLETE</span>`;

const OEE_INPUTS = [
  {id:'T_planned', label:'Planned Production Time',      sym:'T_p',    unit:'h',         conf:'EXACT', unc:'±0.01 h', min:0.5,max:744,  hint:'Per shift/day/week — 8h=1 shift'},
  {id:'T_down',    label:'Unplanned Downtime (Failures)',sym:'T_down', unit:'h',         conf:'STRONG',unc:'±5%',     min:0,            hint:'Big Loss 1: Equipment failures'},
  {id:'T_setup',   label:'Setup & Changeover Time',     sym:'T_setup',unit:'h',         conf:'STRONG',unc:'±5%',     min:0,            hint:'Big Loss 2: Setup & adjustment'},
  {id:'CT_ideal',  label:'Ideal Cycle Time (design)',   sym:'CT_0',   unit:'h/unit',    conf:'EXACT', unc:'±1%',     min:0.0001,       hint:'OEM/design value — NOT measured cycle time'},
  {id:'N_total',   label:'Total Parts Produced',        sym:'N_tot',  unit:'units',     conf:'EXACT', unc:'±0',      min:1,            hint:'All parts including defects'},
  {id:'N_good',    label:'Good Parts (First Pass)',     sym:'N_good', unit:'units',     conf:'EXACT', unc:'±0',      min:0,            hint:'Conforming parts only — no rework'},
  {id:'CM_unit',   label:'Contribution Margin / Unit',  sym:'CM',     unit:'USD/unit',  conf:'STRONG',unc:'±5%',     min:0,            hint:'Revenue − variable cost per unit'},
];

let oeeInp = {};

function initOEE(){
  const g = document.getElementById('oee-inp-grid');
  g.innerHTML = OEE_INPUTS.map(inp=>{
    const cc = inp.conf==='EXACT'?'conf-exact':inp.conf==='STRONG'?'conf-strong':'conf-approx';
    return `<div class="inp-item" data-id="${inp.id}">
      <div class="inp-lbl">${inp.label}<span class="sym">${inp.sym}</span><span class="req">*</span><span class="conf-badge ${cc}">${inp.conf}</span></div>
      <div class="inp-row">
        <input type="number" step="any" placeholder="${inp.hint||''}"
          oninput="oeeInp['${inp.id}']=this.value===''?undefined:+this.value">
        <span class="inp-unit">${inp.unit}</span>
      </div>
      <div class="inp-meta">
        <span class="inp-hint">≥${inp.min}${inp.max?' ≤'+inp.max:''} ${inp.unit}</span>
        <span class="inp-unc">${inp.unc}</span>
      </div>
    </div>`;
  }).join('');
}

function oeeCalc(){
  const i = oeeInp;
  const errs = [];
  const Tp=+(i.T_planned||0), Td=+(i.T_down||0), Ts=+(i.T_setup||0);
  const CT=+(i.CT_ideal||0), Nt=+(i.N_total||0), Ng=+(i.N_good||0), CM=+(i.CM_unit||0);

  if(!Tp||Tp<0.5) errs.push({id:'V0',msg:'Planned production time (T_p) minimum 0.5 h required.'});
  if(!CT||CT<=0)  errs.push({id:'V3',msg:'Ideal cycle time (CT_0) must be positive — V3 BLOCK.'});
  if(!Nt||Nt<1)  errs.push({id:'V0',msg:'Total parts (N_tot) must be at least 1.'});
  if(Ng>Nt) errs.push({id:'V1',msg:'Good parts (N_good) cannot exceed total parts — V1 BLOCK. Logical impossibility.'});
  const OT = Tp-Td-Ts;
  if(OT<=0) errs.push({id:'V2',msg:'Operating Time ≤ 0 — V2 BLOCK. Downtime + setup exceeds planned time.'});

  const ep = document.getElementById('oee-err-panel');
  if(errs.length){
    ep.style.display='block';
    ep.innerHTML=errs.map(e=>`<div class="err-item"><span class="err-code">${e.id}</span><span class="err-msg">${e.msg}</span></div>`).join('');
    return;
  }
  ep.style.display='none';

  const Av = (OT/Tp)*100;
  const Pf_raw = ((CT*Nt)/OT)*100;
  if(Pf_raw>105) {
    const ep2=document.getElementById('oee-err-panel');
    ep2.style.display='block';
    ep2.innerHTML=`<div class="err-item"><span class="err-code">V4</span><span class="err-msg">Performance > 105%: Ideal cycle time appears too long or production data is inconsistent — V4 WARN. Check CT_0 against OEM specification.</span></div>`;
  }
  const Pf = Math.min(Pf_raw, 100);
  const Qu = (Ng/Nt)*100;
  const OEE = (Av/100)*(Pf/100)*(Qu/100)*100;
  const N_max_theo = Tp/CT;
  const N_lost = N_max_theo - Ng;
  const L_fin = N_lost * CM;
  const DPMO = ((Nt-Ng)/Nt)*1e6;

  const warns=[];
  if(OEE<65) warns.push({sev:'CRITICAL',id:'W1',src:'JIPM TPM World-Class Benchmark',msg:`OEE CRITICAL: ${OEE.toFixed(1)}%. JIPM world-class target is ≥85%. This machine is operating at ${((OEE/85)*100).toFixed(0)}% of benchmark. Potential hidden capacity: ~${((85-OEE)/100*N_max_theo).toFixed(0)} additional units per period. Immediate TPM intervention required.`});
  else if(OEE<85) warns.push({sev:'WARNING',id:'W2',src:'JIPM TPM',msg:`OEE Improvement Opportunity: ${OEE.toFixed(1)}%. World-class benchmark is 85%. Gap of ${(85-OEE).toFixed(1)} percentage points represents approximately ${fmtCurrency((85-OEE)/100*N_max_theo*CM)} of unrealized contribution margin per period.`});
  if(Pf<85) warns.push({sev:'WARNING',id:'W3',src:'ISO 22400-2 Phantom Losses',msg:`Phantom Loss Detected: Performance rate ${Pf.toFixed(1)}%. Machine appears running but micro-stops, speed reductions, or jams are reducing throughput. Deploy MES/IIoT micro-stop logging per ANSI/ISA-18.2.`});
  if(Qu<98) warns.push({sev:'WARNING',id:'W4',src:'Six Sigma DPMO',msg:`Quality Loss: FPY = ${Qu.toFixed(2)}% → DPMO = ${DPMO.toFixed(0)}. Scrap + rework cost estimate: ~${fmtCurrency((Nt-Ng)*CM*0.3)}. SPC/Cpk analysis and Poka-yoke recommended.`});
  if(Av<90) warns.push({sev:'INFO',id:'W5',src:'OEE Foundation',msg:`Availability ${Av.toFixed(1)}%: Downtime ${Td.toFixed(1)}h + Setup ${Ts.toFixed(1)}h = ${(Td+Ts).toFixed(1)}h. Planned maintenance (PM) and SMED can significantly improve this metric.`});

  const wHtml=warns.map(w=>`<div class="warn-card ${w.sev}">
    <div class="warn-hdr"><span>${w.sev==='CRITICAL'?'⛔':w.sev==='WARNING'?'⚠':'ℹ'}</span>
    <span class="warn-sev">${w.sev}</span><span class="warn-id">${w.id}</span>
    <span class="warn-src">${w.src}</span></div>
    <div class="warn-msg">${w.msg}</div></div>`).join('');

  const oeeC=OEE>=85?'ok':OEE>=65?'warn':'crit';
  const AvC=Av>=90?'ok':Av>=80?'warn':'crit';
  const PfC=Pf>=90?'ok':Pf>=80?'warn':'crit';
  const QuC=Qu>=99?'ok':Qu>=97?'warn':'crit';

  const scoreCards=`<div class="score-cards">
    <div class="score-card"><div class="sc-label">OEE</div><div class="sc-value ${oeeC}">${OEE.toFixed(1)}<span style="font-size:14px">%</span></div><div class="sc-unit">Overall</div><div class="sc-bench">JIPM ≥ 85%</div></div>
    <div class="score-card"><div class="sc-label">Availability</div><div class="sc-value ${AvC}">${Av.toFixed(1)}<span style="font-size:14px">%</span></div><div class="sc-unit">A</div><div class="sc-bench">Target ≥ 90%</div></div>
    <div class="score-card"><div class="sc-label">Performance</div><div class="sc-value ${PfC}">${Pf.toFixed(1)}<span style="font-size:14px">%</span></div><div class="sc-unit">P</div><div class="sc-bench">Target ≥ 95%</div></div>
    <div class="score-card"><div class="sc-label">Quality</div><div class="sc-value ${QuC}">${Qu.toFixed(2)}<span style="font-size:14px">%</span></div><div class="sc-unit">Q</div><div class="sc-bench">Target ≥ 99%</div></div>
    <div class="score-card"><div class="sc-label">Financial Loss</div><div class="sc-value crit" style="font-size:18px">${fmtCurrency(L_fin)}</div><div class="sc-unit">vs. Max Capacity</div><div class="sc-bench">DPMO: ${DPMO.toFixed(0)}</div></div>
  </div>`;

  const OEEgaugeFill = Math.min(OEE,100);
  const OEEgaugeC = OEE>=85?'#059669':OEE>=65?'#D97706':'#DC2626';
  const oeeGauge=`<div class="uc-gauge-wrap">
    <div class="uc-gauge-hdr"><span class="uc-label">OEE Score vs JIPM Benchmark (85%)</span><span class="uc-value-display ${oeeC}">${OEE.toFixed(1)}%</span></div>
    <div class="uc-gauge-track" style="height:10px">
      <div class="uc-gauge-fill" style="width:${OEEgaugeFill}%;background:${OEEgaugeC}"></div>
      <div style="position:absolute;top:0;left:85%;height:100%;width:1px;background:#1A1915;"></div>
    </div>
    <div class="uc-gauge-zones"><span class="uc-zone">0%</span><span class="uc-zone" style="text-align:left;padding-left:${Math.max(OEEgaugeFill-12,0)}%">← You</span><span class="uc-zone" style="text-align:center">85% JIPM ↑</span><span class="uc-zone">100%</span></div>
  </div>`;

  const rows=[
    {name:'Operating Time',sym:'OT',v:OT,u:'h',d:2},
    {name:'Availability',sym:'A',v:Av,u:'%',d:1,hl:true},
    {name:'Performance',sym:'P',v:Pf,u:'%',d:1,hl:true},
    {name:'Quality (FPY)',sym:'Q',v:Qu,u:'%',d:2,hl:true},
    {name:'OEE',sym:'OEE',v:OEE,u:'%',d:1,hl:true},
    {name:'Theoretical Max Capacity',sym:'N_max',v:N_max_theo,u:'units',d:0},
    {name:'Lost Units (all losses)',sym:'N_lost',v:N_lost,u:'units',d:0},
    {name:'Financial Loss',sym:'L_fin',v:L_fin,u:'USD',d:0,hl:true,curr:true},
    {name:'DPMO',sym:'DPMO',v:DPMO,u:'ppm',d:0},
  ];

  const rHtml=rows.map(r=>{
    const dv = r.curr?fmtCurrency(r.v):(r.d===0?r.v.toFixed(0):fmt(r.v,r.d));
    return `<div class="res-row">
      <span class="res-name">${r.name}<span class="res-sym">${r.sym}</span></span>
      <span></span>
      <span class="res-val${r.hl?' hl':''}">${dv}</span>
      <span class="res-unit">${r.curr?'':r.u}</span>
    </div>`;
  }).join('');

  const verdict = OEE<65
    ? `<div class="verdict-banner fail"><span class="verdict-icon">⛔</span><div class="verdict-body"><div class="verdict-title">CRITICAL OEE — Immediate TPM Intervention Required</div><div class="verdict-sub">OEE ${OEE.toFixed(1)}% vs. JIPM benchmark 85% · Financial loss: ${fmtCurrency(L_fin)}</div></div></div>`
    : OEE<85 ? `<div class="verdict-banner warn"><span class="verdict-icon">⚠</span><div class="verdict-body"><div class="verdict-title">OEE BELOW WORLD-CLASS — Improvement opportunity identified</div><div class="verdict-sub">OEE ${OEE.toFixed(1)}% · Gap to benchmark: ${(85-OEE).toFixed(1)} pp · Estimated revenue opportunity: ${fmtCurrency((85-OEE)/100*N_max_theo*CM)}</div></div></div>`
    : `<div class="verdict-banner pass"><span class="verdict-icon">✓</span><div class="verdict-body"><div class="verdict-title">WORLD-CLASS OEE ACHIEVED</div><div class="verdict-sub">OEE ${OEE.toFixed(1)}% ≥ JIPM 85% · Continue CI initiatives · DPMO: ${DPMO.toFixed(0)}</div></div></div>`;

  document.getElementById('oee-results-content').innerHTML =
    verdict + scoreCards + oeeGauge +
    (warns.length ? `<div class="warn-panel">${wHtml}</div>` : '<div class="ok-banner">✓ OEE ≥ 85%. JIPM world-class benchmark achieved. Continue Kaizen.</div>') +
    `<div class="sec-lbl">Detailed Results <span class="sec-note">ISO 22400-2 · JIPM TPM</span></div>` +
    `<div class="res-table"><div class="res-hdr"><span>KPI</span><span></span><span>Value</span><span>Unit</span></div>${rHtml}</div>` +
    `<div class="refs-row"><span class="ref-ttl">References:</span><span class="ref-tag">ISO 22400-2:2014</span><span class="ref-tag">JIPM TPM</span><span class="ref-tag">OEE Foundation</span><span class="ref-tag">Six Sigma DPMO</span><span class="ref-tag">ANSI/ISA-18.2</span></div>`;

  const crits=warns.filter(w=>w.sev==='CRITICAL').length;
  const wC=warns.filter(w=>w.sev==='WARNING').length;
  document.getElementById('oee-status').innerHTML=crits?`<span class="badge badge-crit">⛔ ${crits} CRITICAL</span>`:wC?`<span class="badge badge-warn">⚠ ${wC} WARNING</span>`:`<span class="badge badge-ok">✓ WORLD-CLASS</span>`;
  document.getElementById('oee-uc-display').innerHTML=`OEE: <strong>${OEE.toFixed(1)}%</strong> &nbsp;|&nbsp; Loss: <strong>${fmtCurrency(L_fin)}</strong>`;
  if(warns.length) document.getElementById('oee-dot').style.display='inline-block';

  const ts=nowISO();
  const inputHash=hashStr(JSON.stringify(oeeInp));
  document.getElementById('oee-audit-body').innerHTML =
    buildAuditRow('tool_id','PRO_098')+
    buildAuditRow('version','1.0.0')+
    buildAuditRow('timestamp',ts)+
    buildAuditRow('design_standard','ISO 22400-2:2014 / JIPM TPM')+
    buildAuditRow('input_hash',inputHash)+
    buildAuditRow('T_planned_h',Tp)+
    buildAuditRow('T_down_h',Td)+
    buildAuditRow('T_setup_h',Ts)+
    buildAuditRow('CT_ideal_h',CT)+
    buildAuditRow('N_total',Nt)+
    buildAuditRow('N_good',Ng)+
    buildAuditRow('CM_unit_USD',CM)+
    buildAuditRow('OT_h',OT.toFixed(2))+
    buildAuditRow('Availability_%',Av.toFixed(1))+
    buildAuditRow('Performance_%',Pf.toFixed(1))+
    buildAuditRow('Quality_FPY_%',Qu.toFixed(2))+
    buildAuditRow('OEE_%',OEE.toFixed(1))+
    buildAuditRow('N_lost',N_lost.toFixed(0))+
    buildAuditRow('Financial_Loss_USD',fmtCurrency(L_fin))+
    buildAuditRow('DPMO',DPMO.toFixed(0))+
    buildAuditRow('JIPM_benchmark_met',OEE>=85?'YES — ≥85%':'NO — Below 85%',OEE>=85?'ok':'fail')+
    buildAuditRow('disclaimer','MES/ERP data validation recommended. Contribution margin must be confirmed with accounting.');

  document.getElementById('oee-cbar-status').innerHTML=crits?`<span class="badge badge-crit">⛔ ${crits} CRITICAL</span>`:wC?`<span class="badge badge-warn">⚠ ${wC} WARNING</span>`:`<span class="badge badge-ok">✓ OEE ≥ 85%</span>`;
  document.getElementById('oee-cbar').style.display='flex';
  document.getElementById('oee-reset-btn').style.display='inline-block';
}

function oeeReset(){
  oeeInp={};
  document.getElementById('oee-err-panel').style.display='none';
  document.getElementById('oee-cbar').style.display='none';
  document.getElementById('oee-reset-btn').style.display='none';
  document.getElementById('oee-status').innerHTML='<span class="badge badge-idle">— NOT CALCULATED</span>';
  document.getElementById('oee-uc-display').innerHTML='';
  document.getElementById('oee-dot').style.display='none';
  document.getElementById('oee-results-content').innerHTML='<div class="empty-state"><div class="empty-state-icon">📊</div>No calculation performed.</div>';
  document.getElementById('oee-audit-body').innerHTML='<div class="audit-empty">No calculation performed.</div>';
  initOEE();
  oeeTab('inputs');
}

function oeeTab(name,el){
  ['inputs','results','formulas','fmea','audit'].forEach(t=>{
    const p=document.getElementById('oee-pane-'+t);
    if(p) p.style.display=t===name?'block':'none';
  });
  document.querySelectorAll('#tool-oee .tabs .tab-btn').forEach(b=>b.classList.remove('active'));
  if(el){el.classList.add('active');}
  else{
    const idx=['inputs','results','formulas','fmea','audit'].indexOf(name);
    const btns=document.querySelectorAll('#tool-oee .tabs .tab-btn');
    if(btns[idx]) btns[idx].classList.add('active');
  }
}

function switchTool(t){
  ['cnc','weld','oee'].forEach(id=>{
    document.getElementById('tool-'+id).style.display=id===t?'block':'none';
  });
  document.querySelectorAll('.ts-btn').forEach((b,i)=>{
    b.classList.toggle('active',['cnc','weld','oee'][i]===t);
  });
}
