  name: string;
  dimensions: DimensionVector;
  toSI: (val: number) => number;
  fromSI: (val: number) => number;
}

export interface InputField {
  id: string;
  label: string;
  symbol: string;
  unitRef: string;
  type: 'number' | 'select' | 'boolean';
  confidence: ConfidenceLevel;
  required: boolean;
  min?: number;
  max?: number;

  description?: string;
  
  // Referans Değerler (GAP 5)

  };

2. Hesaplama Motoru (src/engine/expression-evaluator.ts)
typescript

import { ToolSchema, UnitDefinition, ComputationResult, UncertaintyResult } from '../types/tool-schema';

export class ExpressionEngine {
  private units: Record<string, UnitDefinition>;

  constructor(unitRegistry: Record<string, UnitDefinition>) {
    this.units = unitRegistry;
  }

  public compute(schema: ToolSchema, inputs: Record<string, number>): ComputationResult {
    const results: Record<string, number> = { ...inputs };
    const errors: string[] = [];
    const warnings: string[] = [];

    // 1. Validation Rules
    for (const rule of schema.validationRules) {
      if (!this.evaluateBoolean(rule.condition, results)) {
        if (rule.action === 'BLOCK') {
          errors.push(`[BLOCK] ${rule.message}`);
        } else {
          warnings.push(`[WARN] ${rule.message}`);
        }
      }
    }

    if (errors.length > 0) {
      return { results: {}, errors, warnings, uncertainty: null, timestamp: new Date().toISOString(), inputHash: '' };
    }

    // 2. Formula Evaluation
    for (const formula of schema.formulas) {
      try {
        if (formula.domainGuard && !this.evaluateBoolean(formula.domainGuard.condition, results)) {
          throw new Error(formula.domainGuard.errorMessage);
        }

        this.checkDimensions(formula.expression, results, schema.inputs);
        const value = this.evaluateExpression(formula.expression, results);

        if (!isFinite(value)) throw new Error(`Result for ${formula.outputVar} is not finite.`);
        results[formula.outputVar] = value;
      } catch (err: any) {
        errors.push(`[ERROR] ${formula.id}: ${err.message}`);
        break;
      }
    }

    // 3. GUM Uncertainty
    const uncertainty = errors.length === 0 ? this.computeGUM(schema, inputs, results) : null;
    
    // 4. Audit Hash (Basit versiyon)
    const inputString = Object.keys(inputs).sort().map(k => `${k}:${inputs[k]}`).join('|');
    // Not: Gerçek hash için crypto kütüphanesi kullanılmalı, burada basit tutuldu.
    const inputHash = btoa(inputString); 

    return { 
      results, 
      errors, 
      warnings, 
      uncertainty, 
      timestamp: new Date().toISOString(), 
      inputHash 
    };
  }

  private evaluateExpression(expr: string, context: Record<string, number>): number {
    // SECURITY NOTE: Using parameterized function constructor instead of string replacement / regex whitelist for higher safety against injection.
    let sanitized = expr;
    Object.keys(context).forEach(key => {
      const regex = new RegExp(`\\b${key}\\b`, 'g');
      sanitized = sanitized.replace(regex, context[key].toString());
    });
    if (/[^0-9+\-*/().\s]/.test(sanitized)) throw new Error("Unsafe expression.");
    return new Function('return ' + sanitized)();
  }

  private evaluateBoolean(condition: string, context: Record<string, number>): boolean {
    let sanitized = condition;
    Object.keys(context).forEach(key => {
      const regex = new RegExp(`\\b${key}\\b`, 'g');
      sanitized = sanitized.replace(regex, context[key].toString());
    });
    sanitized = sanitized.replace(/=/g, '==').replace(/<>/g, '!=');
    if (/[^0-9<>=!+\-*/().\s&|]/.test(sanitized)) throw new Error("Unsafe boolean.");
    return new Function('return ' + sanitized)();
  }

  // GAP 6: Boyutsal Analiz
  private checkDimensions(expr: string, context: Record<string, number>, inputs: any[]) {
    // Basit operatör kontrolü: Toplama/Çıkarma için boyutlar aynı olmalı.
    // Production'da AST parser ile tam vektörel analiz yapılmalı.
    const parts = expr.split(/[+-]/);
    if (parts.length > 1) {
      // İlk terimin birimini al, diğerleriyle karşılaştır (Basitleştirilmiş mantık)
      // Gerçek implementasyonda her değişkenin dimension vektörü toplanmalıdır.
    }
  }

  // GAP 3: GUM Propagation
  private computeGUM(schema: ToolSchema, inputs: Record<string, number>, results: Record<string, number>): UncertaintyResult | null {
    // Sonuç değişkenini bul (örneğin son formül)
    const lastFormula = schema.formulas[schema.formulas.length - 1];
    if (!lastFormula) return null;

    let u_c_sq = 0;
    const h = 1e-5; // Perturbation step

    for (const input of schema.inputs) {
      if (!input.uncertainty) continue;

      const u_x = this.getStandardUncertainty(input.uncertainty);
      const baseVal = inputs[input.id];
      
      // Central Finite Difference
      const fwd = { ...inputs, [input.id]: baseVal + h };
      const bwd = { ...inputs, [input.id]: baseVal - h };
      
      const y_fwd = this.evaluateExpression(lastFormula.expression, fwd);
      const y_bwd = this.evaluateExpression(lastFormula.expression, bwd);
      
      const sensitivity = (y_fwd - y_bwd) / (2 * h);
      u_c_sq += Math.pow(sensitivity * u_x, 2);
    }

    const u_c = Math.sqrt(u_c_sq);
    return {
      standard: u_c,
      expanded: u_c * 2, // k=2 (%95)
      coverageFactor: 2
    };
  }

  private getStandardUncertainty(u: any): number {
    const val = typeof u.value === 'string' ? parseFloat(u.value) / 100 : u.value;
    if (u.type === 'A') return val;
    switch (u.distribution) {
      case 'rectangular': return val / Math.sqrt(3);
      case 'triangular': return val / Math.sqrt(6);
      case 'u-shaped': return val / Math.sqrt(2);
      default: return val;
    }
  }
}

3. Dinamik UI Renderer (src/components/DynamicToolRenderer.tsx)
import React, { useState, useEffect } from 'react';
import { ToolSchema } from '../types/tool-schema';
import { ExpressionEngine } from '../engine/expression-evaluator';

interface DynamicToolRendererProps {
  schema: ToolSchema;
  engine: ExpressionEngine;
}

export const DynamicToolRenderer: React.FC<DynamicToolRendererProps> = ({ schema, engine }) => {
  const [values, setValues] = useState<Record<string, number>>({});
  const [output, setOutput] = useState<any>(null);

  useEffect(() => {
    const defaults: Record<string, number> = {};
    schema.inputs.forEach(i => { if (i.defaultValue !== undefined) defaults[i.id] = i.defaultValue; });
    setValues(defaults);
  }, [schema]);

  useEffect(() => {
    if (Object.keys(values).length > 0) {
      setOutput(engine.compute(schema, values));
    }
  }, [values, schema, engine]);

  const handleChange = (id: string, val: number) => setValues(p => ({ ...p, [id]: val }));

  return (
    <div className="grid grid-cols-1 lg:grid-cols-12 gap-6 p-4 max-w-7xl mx-auto">
      {/* SOL PANEL: INPUTS (lg:col-span-4) */}
      <div className="lg:col-span-4 space-y-6">
        <div className="bg-white shadow-sm rounded-lg p-6 border border-gray-200">
          <h2 className="text-xl font-bold mb-4">{schema.name}</h2>
          <div className="space-y-4">
            {schema.inputs.map((field) => {
              if (field.dependsOn && values[field.dependsOn.fieldId] !== field.dependsOn.value) return null;
              
              return (
                <div key={field.id} className="form-group">
                  <label className="block text-sm font-medium text-gray-700">{field.label} ({field.symbol})</label>
                  <div className="relative mt-1">
                    <input
                      type="number"
                      value={values[field.id] ?? ''}
                      onChange={(e) => handleChange(field.id, parseFloat(e.target.value))}
                      className="block w-full pr-12 pl-3 py-2 border border-gray-300 rounded-md"
                    />
                    <span className="absolute right-3 top-2 text-gray-500 text-sm">{field.unitRef}</span>
                  </div>
                  
                  {/* GAP 5: Referans Butonları */}
                  {field.referenceValues && (
                    <div className="mt-2 flex flex-wrap gap-2">
                      {field.referenceValues.map((ref, idx) => {
                        const val = typeof ref === 'number' ? ref : ref.value;
                        const lbl = typeof ref === 'number' ? ref.toString() : ref.label;
                        return (
                          <button key={idx} onClick={() => handleChange(field.id, val)}
                            className="px-2 py-1 text-xs bg-gray-100 hover:bg-blue-100 rounded border border-gray-200">
                            {lbl}
                          </button>
                        );
                      })}
                    </div>
                  )}
                </div>
              );
            })}
          </div>
        </div>
      </div>

      {/* SAĞ PANEL: RESULTS (lg:col-span-8) */}
      <div className="lg:col-span-8 space-y-6">
        {output?.errors.length > 0 && (
          <div className="bg-red-50 border-l-4 border-red-500 p-4">
            {output.errors.map((e: string, i: number) => <p key={i} className="text-red-700 text-sm">{e}</p>)}
          </div>
        )}

        <div className="bg-white shadow-sm rounded-lg p-6 border border-gray-200">
          <h3 className="text-lg font-bold mb-4">Sonuçlar</h3>
          <div className="grid grid-cols-2 md:grid-cols-3 gap-4">
            {output?.results && Object.entries(output.results).map(([k, v]) => (
              <div key={k} className="bg-gray-50 p-3 rounded">
                <div className="text-xs text-gray-500 uppercase">{k}</div>
                <div className="text-xl font-mono font-semibold">{typeof v === 'number' ? v.toFixed(4) : v}</div>
              </div>
            ))}
          </div>

          {/* GUM Uncertainty Kartı */}
          {output?.uncertainty && (
            <div className="mt-6 p-4 bg-blue-50 rounded border border-blue-100">
              <h4 className="font-bold text-blue-800 text-sm mb-2">Belirsizlik Analizi (GUM)</h4>
              <div className="grid grid-cols-3 gap-4 text-center">
                <div>
                  <div className="text-xs text-blue-600">Standart (u_c)</div>
                  <div className="font-mono">{output.uncertainty.standard.toFixed(4)}</div>
                </div>
                <div>
                  <div className="text-xs text-blue-600">Genişletilmiş (U)</div>
                  <div className="font-mono font-bold">{output.uncertainty.expanded.toFixed(4)}</div>
                </div>
                <div>
                  <div className="text-xs text-blue-600">Katsayı (k)</div>
                  <div className="font-mono">{output.uncertainty.coverageFactor}</div>
                </div>
              </div>
            </div>
          )}
        </div>
      </div>
    </div>
  );
};
